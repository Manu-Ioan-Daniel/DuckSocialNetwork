CREATE TABLE users(
	id SERIAL PRIMARY KEY,
)

CREATE TABLE friendships(
	user_id_1 BIGINT REFERENCES users(id) ON DELETE CASCADE,
	user_id_2 BIGINT REFERENCES users(id) ON DELETE CASCADE,
	PRIMARY KEY (user_id_1,user_id_2), 
	CHECK (user_id_1<user_id_2)
);

CREATE TABLE friend_requests(
	from_id BIGINT REFERENCES users(id) ON DELETE CASCADE,
	to_id BIGINT REFERENCES users(id) ON DELETE CASCADE,
	PRIMARY KEY (from_id,to_id),
	status VARCHAR(100) NOT NULL,
	CHECK (status IN ('pending','accepted','denied'))
);

CREATE  UNIQUE INDEX unique_fr_pair ON friend_requests(
	LEAST(from_id,to_id),
	GREATEST(from_id,to_id)
);

CREATE TABLE messages(
	id SERIAL PRIMARY KEY,
	fromId BIGINT REFERENCES users(id) ON DELETE CASCADE,
	toId BIGINT REFERENCES users(id) ON DELETE CASCADE,
	message VARCHAR(100) NOT NULL,
	date TIMESTAMP NOT NULL
);

CREATE TABLE reply_messages(
	id SERIAL PRIMARY KEY,
	message_id BIGINT REFERENCES messages(id) ON DELETE CASCADE
);

CREATE TABLE events(
	id SERIAL PRIMARY KEY,
	name VARCHAR(100) NOT NULL UNIQUE
	
);

CREATE TABLE event_members(
	user_id BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
	event_id BIGINT NOT NULL REFERENCES events(id) ON DELETE CASCADE,
	date TIMESTAMP DEFAULT NOW(),
	PRIMARY KEY (user_id,event_id)
);

CREATE TABLE notifications(
	id SERIAL PRIMARY KEY,
	user_id BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
	message VARCHAR(100) NOT NULL
);
